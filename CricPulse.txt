function init(){

function setLeague(l,btn){
  curLeague=l;
  document.querySelectorAll('.ltab').forEach(b=>b.classList.remove('active'));
  btn.classList.add('active');
  document.getElementById('teamSearch').value='';
  showGrid(TEAMS[l].map(t=>({...t,lg:l})));
}

function filterTeams(){
  const q=document.getElementById('teamSearch').value.toLowerCase().trim();
  if(!q){showGrid(TEAMS[curLeague].map(t=>({...t,lg:curLeague})));return;}
  showGrid(ALL.filter(t=>t.n.toLowerCase().includes(q)||(t.s||'').toLowerCase().includes(q)));
}

function showGrid(teams){
  curFiltered=teams;
  const g=document.getElementById('teamGrid');
  g.innerHTML=teams.map(t=>{
    const isBat=sel[0]&&sel[0].n===t.n;
    const isBowl=sel[1]&&sel[1].n===t.n;
    const isFav=fav&&fav.n===t.n;
    let cls='tcard'+(isBat?' bat':isBowl?' bowl':'')+(isFav?' fav':'');
    let badge=isBat?`<span class="tbadge bat-b">BAT</span>`:isBowl?`<span class="tbadge bowl-b">BOWL</span>`:'';
    if(isFav)badge+=`<span class="tbadge fav-b" style="top:${isBat||isBowl?'18px':'4px'}">❤️</span>`;
    return `<div class="${cls}" onclick="pickTeam('${esc(t.n)}','${t.f}','${t.lg}','${t.s||''}')">
      <span class="tflag">${t.f}</span>
      <div class="tinfo"><div class="tname">${t.n}</div><div class="tleague">${t.lg}</div></div>${badge}
    </div>`;
  }).join('');
}

function esc(s){return s.replace(/'/g,"\\'");}

function pickTeam(name,flag,lg,s){
  if(sel[0]&&sel[1]){
    if(sel[0].n===name){sel[0]=null;}
    else if(sel[1].n===name){sel[1]=null;}
    else{
      if(fav&&sel[0].n===fav.n)sel[1]={n:name,f:flag,lg,s};
      else sel[0]={n:name,f:flag,lg,s};
    }
  } else if(!sel[0]){sel[0]={n:name,f:flag,lg,s};}
  else if(sel[0].n===name){sel[0]=null;if(sel[1]){sel[0]=sel[1];sel[1]=null;}}
  else{sel[1]={n:name,f:flag,lg,s};}
  if(fav&&!sel.find(t=>t&&t.n===fav.n))fav=null;
  showGrid(curFiltered);
  updateSel();
}

function updateSel(){
  const both=sel[0]&&sel[1];
  const any=sel[0]||sel[1];
  document.getElementById('selRowWrap').style.display=any?'block':'none';
  document.getElementById('s0f').textContent=sel[0]?sel[0].f:'🏏';
  document.getElementById('s0n').textContent=sel[0]?sel[0].n:'Pick team 1';
  document.getElementById('s0n').style.color=sel[0]?'var(--text)':'var(--muted)';
  document.getElementById('s1f').textContent=sel[1]?sel[1].f:'⚾';
  document.getElementById('s1n').textContent=sel[1]?sel[1].n:'Pick team 2';
  document.getElementById('s1n').style.color=sel[1]?'var(--text)':'var(--muted)';
  document.getElementById('favSec').style.display=both?'block':'none';
  if(both){
    const opts=[sel[0],sel[1],{n:'None / Neutral',f:'🤝',s:''}];
    document.getElementById('favBtns').innerHTML=opts.map(t=>{
      const isA=t.n==='None / Neutral'?!fav:(fav&&fav.n===t.n);
      return `<button class="fav-btn${isA?' active':''}" onclick="setFav('${esc(t.n)}','${t.f}')">
        <span class="fb-flag">${t.f}</span>
        <div class="fb-name">${t.n}</div>
        <div class="fb-role">${t.n==='None / Neutral'?'neutral':t.n===sel[0].n?'batting':'bowling'}</div>
        <span class="heart">❤️</span>
      </button>`;
    }).join('');
  }
  document.getElementById('step1Btn').disabled=!both;
}

function setFav(name,flag){
  fav=name==='None / Neutral'?null:{n:name,f:flag};
  updateSel();
}
function swapTeams(){[sel[0],sel[1]]=[sel[1],sel[0]];showGrid(curFiltered);updateSel();}

function parseOv(o){const p=String(o).split('.');return(+p[0]||0)*6+Math.min(+(p[1]||'0').charAt(0)||0,5);}
function ovStr(b){return`${Math.floor(b/6)}.${b%6}`;}
function getFmt(){const a=document.querySelector('.fmt-btn.active');if(!a)return'T20';const n=a.querySelector('.fmt-name').textContent.trim();return n==='100'?'The Hundred':n==='Custom'?'Custom':n;}
function dots(done,total,cls){
  const totOv=Math.ceil(total/6),dOv=Math.floor(done/6),pb=done%6;
  let d='';
  for(let i=0;i<Math.min(totOv,50);i++){
    if(i<dOv)d+=`<div class="dot done"></div>`;
    else if(i===dOv&&pb>0)d+=`<div class="dot part"></div>`;
    else d+=`<div class="dot ${cls}"></div>`;
  }
  return d;
}
function matchHdr(bat,bowl,sc,wk,bd,extra,fmt,inn){
  const fb=fav&&bat&&fav.n===bat.n,fbw=fav&&bowl&&fav.n===bowl.n;
  return`<div class="match-card">

    <div class="match-teams">
      <div class="mt">
        <span class="mt-flag">${bat?bat.f:'🏏'}</span>
        <div><div class="mt-name">${bat?bat.n:'Batting'}</div>
        <div class="mt-score">${sc}${wk!==null?'/'+wk:''} (${ovStr(bd)})</div>

        </div>

      <div class="mt right">
        <span class="mt-flag">${bowl?bowl.f:'⚾'}</span>
        <div><div class="mt-name">${bowl?bowl.n:'Bowling'}</div>

        ${fbw?`<div class="mt-heart">❤️ your team</div>`:''}

      </div>

function favMsg(winPct,isBatting){

}

  const wk=+document.getElementById('i1wk').value||0;

  const totalBalls=isHundred?100:totalOvers*6;
  const bd=parseOv(ovIn);
  const bl=totalBalls-bd;
  const crr=bd>0?(sc/bd*6):0;
  const wkLeft=10-wk;

  // Projection
  const wkF=Math.max(0.5,1-(wk*0.045));
  const prog=bd/totalBalls;
  const acc=prog<0.3?1.13:prog<0.6?1.06:0.97;
  const proj=Math.round(sc+(bl/6)*crr*wkF*acc);
  const projLow=Math.round(proj*0.88),projHigh=Math.round(proj*1.12);

  // Win probability
  const fmtAvg={20:160,10:100,50:265,100:150}[isHundred?100:totalOvers]||160;
  const winP=Math.max(10,Math.min(90,50+(proj-fmtAvg)*0.55));

  // Pace
  const avgRR={20:8.0,10:9.0,50:5.5,100:8.5}[isHundred?100:totalOvers]||8.0;
  let pC='a',pL='On Par',pD='Scoring at an average pace.';
  if(crr>=avgRR*1.15){pC='g';pL='Above Average';pD='Ahead of the typical scoring rate.';}
  else if(crr<=avgRR*0.85){pC='r';pL='Below Par';pD='Behind — acceleration needed in later overs.';}

  const bat=sel[0],bowl=sel[1],fmt=getFmt();
  let html=matchHdr(bat,bowl,sc,wk,bd,`${bl} balls left`,fmt,1);

  html+=`<div class="match-card"><div class="stats-row">

    <div class="stat"><div class="stat-v">${wkLeft}</div><div class="stat-l">Wkts Left</div><div class="stat-s">${wk} fallen</div></div>
  </div></div>`;

  const projCC=proj>=180?'easy':proj>=140?'tough':'hard';
  html+=`<div class="rrr-card ${projCC}">
    <div class="rrr-top">
      <div><div class="rrr-meta">Projected Final Score</div><div class="rrr-num">${proj}</div><div style="font-size:11px;color:var(--muted);margin-top:3px">Range: ${projLow} – ${projHigh}</div></div>
      <div class="rrr-right"><span class="rrr-status">${pL}</span><p class="rrr-desc">${pD}</p></div>
    </div>
    <div class="prog-track"><div class="prog-fill" id="pf1" style="width:0%"></div></div>
    <div class="prog-labs"><span>120</span><span>160</span><span>200+ 🔥</span></div>
  </div>`;

  html+=winBar(winP,bat?bat.n:'Batting',bowl?bowl.n:'Bowling');
  html+=favMsg(winP,fav&&bat&&fav.n===bat.n);

  // Par tracker
  let parH='';
  if(par>0){
    const parNow=Math.round((par/totalBalls)*bd);
    const diff=sc-parNow;
    parH=`<div class="mini" style="border-color:${diff>=0?'rgba(74,222,128,.3)':'rgba(248,113,113,.3)'}">
      <div class="mini-l">vs Par (${par})</div>
      <div class="mini-v" style="color:${diff>=0?'var(--green)':'var(--red)'}">${diff>=0?'+':''}${diff}</div>
      <div class="mini-s">Par at this point: ${parNow} · ${Math.abs(diff)} runs ${diff>=0?'ahead':'behind'}</div>
    </div>`;
  } else {
    parH=`<div class="mini"><div class="mini-l">Runs to 200</div>
      <div class="mini-v" style="color:${proj>=200?'var(--green)':'var(--amber)'}">${Math.max(0,200-proj)}</div>
      <div class="mini-s">needed in projected total</div></div>`;
  }

  const msH=[100,150,180,200].filter(m=>m>sc).slice(0,3).map(m=>{
    const rr=bl>0?((m-sc)/bl*6):Infinity;
    const c=rr<=8?'var(--green)':rr<=13?'var(--amber)':'var(--red)';
    return`<div class="ms-row"><span style="color:var(--muted2)">Reach ${m}</span>
      <span style="color:${c};font-family:'DM Mono',monospace;font-size:11px">${isFinite(rr)?rr.toFixed(1)+' rpo':'✓'}</span></div>`;
  }).join('');

  const d=dots(bd,totalBalls,'left-b');
  html+=`<div class="bot">${parH}
    <div class="mini"><div class="mini-l">Scoring milestones</div><div style="margin-top:6px">${msH}</div></div>
  </div>`;
  html+=`<div class="mini" style="margin-bottom:12px"><div class="mini-l">Innings progress</div><div class="dots">${d}</div>
    <div class="dot-leg"><span><i style="background:rgba(255,255,255,.08)"></i>bowled</span><span><i style="background:var(--amber);opacity:.85"></i>current</span><span><i style="background:var(--blue);opacity:.55"></i>remaining</span></div>
  </div>`;

  show(html);
  setTimeout(()=>{const pf=document.getElementById('pf1');if(pf)pf.style.width=Math.min((proj/220)*100,100)+'%';},60);
  animBars(winP);

function parseOvAPI(s){
  if(!s)return 0;
  const p=String(s).trim().split('.');
  return(+p[0]||0)*6+Math.min(+(p[1]||'0').charAt(0)||0,5);
}

function matchTeamAPI(apiName){
  if(!apiName)return{n:apiName,f:'🏏',lg:'Live',s:''};
  const q=apiName.toLowerCase();
  for(const t of ALL){
    if(q.includes(t.n.toLowerCase())||t.n.toLowerCase().includes(q.split(' ')[0]))return t;
    if(t.s&&q.includes(t.s.toLowerCase()))return t;
  }
  return{n:apiName,f:'🏏',lg:'Live',s:''};
}

function parseApiMatch(m){
  try{
    const name=m.name||'';
    const parts=name.split(' vs ');
    const t1=matchTeamAPI((parts[0]||'').trim());
    const t2=matchTeamAPI(((parts[1]||'').split(',')[0]).trim());
    const nl=name.toLowerCase();
    let fmt=20;
    if(nl.includes('test'))fmt=90;
    else if(nl.includes('odi')||nl.includes('50 over'))fmt=50;
    else if(nl.includes('t10'))fmt=10;
    else if(nl.includes('100-ball'))fmt=100;
    const scores=m.score||[];
    let inn=1,sc1=0,wk1=0,ov1=0,sc2=0,wk2=0,ov2=0,target=0;
    if(scores.length>=1){sc1=scores[0].r||0;wk1=scores[0].w||0;ov1=parseOvAPI(scores[0].o);}
    if(scores.length>=2){inn=2;sc2=scores[1].r||0;wk2=scores[1].w||0;ov2=parseOvAPI(scores[1].o);target=sc1+1;}
    return{t1,t2,fmt,inn,sc1,wk1,ov1,sc2,wk2,ov2,target,status:m.status||'',name};
  }catch(e){return null;}
}

async function fetchAllMatches(){
  const res=await fetch(`https://api.cricapi.com/v1/currentMatches?apikey=${CRIC_API_KEY}&offset=0`);
  const data=await res.json();
  if(data&&data.status==='success'&&data.data){
    allLiveMatches=data.data.filter(m=>m.matchStarted&&!m.matchEnded);
    return data.data;
  }
  return[];
}
